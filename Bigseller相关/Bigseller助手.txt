(function () {
    'use strict';

    const STYLE = 'margin-left:6px;padding:2px 8px;border:none;background:#722ed1;color:#fff;border-radius:3px;cursor:pointer;font-size:12px;white-space:nowrap;box-shadow:0 1px 2px rgba(0,0,0,.2);';

    const WAREHOUSE_ID = '54443';
    const CLOUD_BASE = 'https://env-00jy671a213o.dev-hz.cloudbasefunction.cn/bigsellerSwap';

    // GM_xmlhttpRequest 封装为 Promise
    function gmFetch(url, data) {
        return new Promise((resolve, reject) => {
            GM_xmlhttpRequest({
                method: 'POST',
                url,
                headers: { 'Content-Type': 'application/json' },
                data: JSON.stringify(data),
                onload: (resp) => {
                    try { resolve(JSON.parse(resp.responseText)); }
                    catch (e) { reject(e); }
                },
                onerror: (e) => reject(e)
            });
        });
    }

    // 云端：写入换货记录
    async function saveSwapRecord(record) {
        try {
            const json = await gmFetch(CLOUD_BASE + '/addSwapRecord', record);
            return json;
        } catch (e) {
            console.log('[Bigseller助手] 保存换货记录失败:', e.message || e);
            return null;
        }
    }

    // 云端：批量查询换货记录
    async function querySwapRecords(orderNos) {
        try {
            const json = await gmFetch(CLOUD_BASE + '/getSwapRecords', { orderNos });
            if (json.code === 200) return json.data;
            return [];
        } catch (e) {
            console.log('[Bigseller助手] 查询换货记录失败:', e.message || e);
            return [];
        }
    }

    // 通过库存接口查询SKU信息（标题+库存）
    async function fetchSkuInfo(sku) {
        const params = new URLSearchParams({
            pageNo: '1',
            pageSize: '50',
            searchType: 'skuName',
            searchContent: sku,
            inquireType: '2',
            stockStatus: '',
            isGroup: '',
            orderBy: '',
            desc: '',
            fullCid: '',
            queryDistribution: '1',
            saleState: '',
            zoneId: '',
            openFlag: 'false',
            hideZeroInventorySku: '0',
            warehouseIds: WAREHOUSE_ID
        });
        const resp = await fetch('/api/v1/inventory/pageList.json', {
            method: 'POST',
            headers: { 'Content-Type': 'application/x-www-form-urlencoded', 'clienttype': '1' },
            body: params.toString()
        });
        const json = await resp.json();
        if (json.code === 0 && json.data && json.data.page && json.data.page.rows.length) {
            const item = json.data.page.rows[0];
            return { title: item.title || '--', available: item.available ?? '--', image: item.image || '', skuId: item.skuId || null };
        }
        return null;
    }

    // 通过SKU前缀搜索同款SKU列表
    async function fetchRelatedSkus(skuPrefix, excludeSku) {
        const params = new URLSearchParams({
            pageNo: '1',
            pageSize: '500',
            searchType: 'skuName',
            searchContent: skuPrefix,
            inquireType: '1',
            stockStatus: '',
            isGroup: '',
            orderBy: '',
            desc: '',
            fullCid: '',
            queryDistribution: '1',
            saleState: '',
            zoneId: '',
            openFlag: 'false',
            hideZeroInventorySku: '0',
            warehouseIds: WAREHOUSE_ID
        });
        const resp = await fetch('/api/v1/inventory/pageList.json', {
            method: 'POST',
            headers: { 'Content-Type': 'application/x-www-form-urlencoded', 'clienttype': '1' },
            body: params.toString()
        });
        const json = await resp.json();
        if (json.code === 0 && json.data && json.data.page && json.data.page.rows.length) {
            return json.data.page.rows
                .filter(item => item.sku !== excludeSku)
                .map(item => ({ sku: item.sku, title: item.title || '--', available: item.available ?? 0, image: item.image || '' }));
        }
        return [];
    }

    // 从标题中提取尺寸关键词（如 "200*250"）
    function extractSizeKeyword(title) {
        const match = title.match(/(\d+)\s*[\*xX×]\s*(\d+)/);
        return match ? match[0] : null;
    }

    // 从标题中解析图案和尺寸（宽*高）
    // 标题格式: "暗无天日李印花魔术贴;气球熊;150*170"（宽*高）
    function parseTitleInfo(title) {
        const parts = title.split(';');
        const pattern = parts.length >= 2 ? parts[parts.length - 2].trim() : '';
        const sizeMatch = title.match(/(\d+)\s*[\*xX×]\s*(\d+)/);
        const width = sizeMatch ? parseInt(sizeMatch[1]) : 0;
        const height = sizeMatch ? parseInt(sizeMatch[2]) : 0;
        const area = width * height;
        return { pattern, width, height, area };
    }

    // 推荐排序：有库存优先，异图同尺 > 同图更大 > 异图更大
    // 同尺寸判断：宽和高都完全匹配（不是面积相等）
    function sortRecommendations(list, currentTitle) {
        const current = parseTitleInfo(currentTitle);

        function getPriority(info) {
            const sameSize = info.width === current.width && info.height === current.height;
            const samePattern = info.pattern === current.pattern;
            const largerSize = info.area > current.area;
            if (sameSize && !samePattern) return { level: 1, label: '异图同尺' };
            if (samePattern && largerSize) return { level: 2, label: '同图更大' };
            if (!samePattern && largerSize) return { level: 3, label: '异图更大' };
            return { level: 4, label: '' };
        }

        return list.map(item => {
            const info = parseTitleInfo(item.title);
            const pri = getPriority(info);
            return { ...item, _priority: pri.level, _label: pri.label, _info: info };
        }).sort((a, b) => {
            const aHasStock = a.available > 0 ? 1 : 0;
            const bHasStock = b.available > 0 ? 1 : 0;
            if (aHasStock !== bHasStock) return bHasStock - aHasStock;
            if (a._priority !== b._priority) return a._priority - b._priority;

            // 同图更大 / 异图更大 内部排序
            if (a._priority === 2 || a._priority === 3) {
                const aSameH = a._info.height === current.height ? 0 : 1;
                const bSameH = b._info.height === current.height ? 0 : 1;
                // 高度相等的优先
                if (aSameH !== bSameH) return aSameH - bSameH;
                // 面积越小越靠前
                if (a._info.area !== b._info.area) return a._info.area - b._info.area;
            }

            return (b.available ?? 0) - (a.available ?? 0);
        });
    }

    // 从SKU中提取款号（如 "JS-017-120" → "JS-017"）
    function extractStylePrefix(sku) {
        const parts = sku.split('-');
        if (parts.length >= 2) return parts.slice(0, 2).join('-');
        return sku;
    }

        mask.innerHTML = `

    <div style="font-size:16px;font-weight:bold;margin-bottom:16px;">换货操作</div>

            <label style="font-size:13px;color:#666;font-weight:bold;">当前SKU</label>

            <div style="margin-top:8px;font-size:14px;word-break:break-all;">${currentSku}</div>

            <div style="margin-top:6px;font-size:12px;color:#999;">库存: ${currentStock}</div>
        </div>
        <div style="display:flex;align-items:center;font-size:20px;color:#999;">→</div>
        <div style="flex:1;padding:12px;border:1px solid #722ed1;border-radius:6px;background:#f9f0ff;">
            <label style="font-size:13px;color:#666;font-weight:bold;">目标SKU</label>
            <img id="c_swap_to_img" src="" style="width:60px;height:60px;object-fit:cover;border-radius:4px;margin-top:8px;display:none;" />
            <div style="margin-top:8px;">
                <input id="c_swap_to" type="text" placeholder="输入目标SKU"
                    style="width:100%;padding:4px 8px;border:1px solid #d9d9d9;border-radius:4px;font-size:13px;" />
            </div>
            <div id="c_swap_to_title" style="margin-top:4px;font-size:12px;color:#666;">标题: --</div>
            <div id="c_swap_to_stock" style="margin-top:6px;font-size:12px;color:#999;">库存: --</div>
        </div>
    </div>
    <div style="margin-bottom:16px;">
        <label style="font-size:13px;color:#666;font-weight:bold;">推荐SKU列表</label>
        <div id="c_swap_recommend" style="margin-top:8px;padding:10px;border:1px solid #e8e8e8;border-radius:4px;min-height:60px;max-height:200px;overflow-y:auto;font-size:12px;color:#999;">
            暂无推荐
        </div>
    </div>
    <div style="display:flex;justify-content:flex-end;gap:10px;">
        <button id="c_swap_cancel" style="padding:6px 16px;border:1px solid #d9d9d9;background:#fff;border-radius:4px;cursor:pointer;font-size:13px;">取消</button>
        <button id="c_swap_confirm" style="padding:6px 16px;border:none;background:#722ed1;color:#fff;border-radius:4px;cursor:pointer;font-size:13px;">确认换货</button>

        // 锁定背景滚动
        document.body.style.overflow = 'hidden';

        function closeModal() {
            mask.remove();
            document.body.style.overflow = '';
        }

        // 点击遮罩或取消关闭
        mask.addEventListener('click', e => { if (e.target === mask) closeModal(); });
        document.getElementById('c_swap_cancel').onclick = () => closeModal();
        document.getElementById('c_swap_confirm').onclick = async () => {
            const targetSku = document.getElementById('c_swap_to').value.trim();
            if (!targetSku) { alert('请输入目标SKU'); return; }
            if (!orderNo) { alert('未获取到订单号'); return; }

            const confirmBtn = document.getElementById('c_swap_confirm');
            confirmBtn.innerText = '换货中...';
            confirmBtn.disabled = true;

            try {
                // 1. 获取目标SKU的skuId
                const targetInfo = await fetchSkuInfo(targetSku);
                if (!targetInfo || !targetInfo.skuId) {
                    alert('未找到目标SKU信息');
                    confirmBtn.innerText = '确认换货';
                    confirmBtn.disabled = false;
                    return;
                }

                // 2. 通过订单号获取orderId
                const orderResp = await fetch('/api/v1/order/new/pageList.json', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json', 'clienttype': '1' },
                    body: JSON.stringify({
                        searchType: 'orderNo',
                        searchContent: orderNo,
                        inquireType: 2,
                        allOrder: true,
                        historyOrder: 0,
                        pageNo: 1,
                        pageSize: 10,
                        desc: 1,
                        orderBy: 'printTime'
                    })
                });
                const orderJson = await orderResp.json();
                if (orderJson.code !== 0 || !orderJson.data?.page?.rows?.length) {
                    alert('未找到订单');
                    confirmBtn.innerText = '确认换货';
                    confirmBtn.disabled = false;
                    return;
                }
                const orderId = orderJson.data.page.rows[0].id;

                // 3. 获取订单详情中的orderItemId
                const detailResp = await fetch(`/api/v1/order/detail.json?id=${orderId}&historyOrder=0&viewBuyerMessage=false`, {

                });

                if (detailJson.code !== 0 || !detailJson.data?.orderDetail?.orderItemVoList?.length) {
                    alert('获取订单详情失败');
                    confirmBtn.innerText = '确认换货';
                    confirmBtn.disabled = false;
                    return;
                }
                const orderItem = detailJson.data.orderDetail.orderItemVoList.find(
                    item => item.varSku === currentSku || item.inventorySku === currentSku
                );
                if (!orderItem) {
                    alert('订单中未找到当前SKU对应的商品项');
                    confirmBtn.innerText = '确认换货';
                    confirmBtn.disabled = false;
                    return;
                }

                // 4. 调用换货接口
                const swapParams = new URLSearchParams({
                    orderId: String(orderId),
                    orderItemId: String(orderItem.id),
                    orderState: 'new',
                    warehouseId: WAREHOUSE_ID,
                    skuId: String(targetInfo.skuId),
                    platformSku: targetSku,
                    changeType: 'only'
                });

                    method: 'POST',
                    headers: { 'Content-Type': 'application/x-www-form-urlencoded', 'clienttype': '1' },

                });

                        orderNo,
                        orderId: String(orderId),
                        originalSku: currentSku,

                        targetSkuId: String(targetInfo.skuId),

                        targetAvailable: targetInfo.available,

                    injectSwapResult(skuTag, targetSku, targetInfo);
                    closeModal();
                } else {
                    alert('换货失败: ' + (swapJson.msg || '未知错误'));
                    confirmBtn.innerText = '确认换货';
                    confirmBtn.disabled = false;
                }
            } catch (e) {
                alert('换货出错: ' + e.message);
                confirmBtn.innerText = '确认换货';
                confirmBtn.disabled = false;
            }
        };

        const toInput = document.getElementById('c_swap_to');
        toInput.addEventListener('blur', async () => {

            if (!targetSku) return;
            const titleEl = document.getElementById('c_swap_to_title');
            const stockEl = document.getElementById('c_swap_to_stock');
            const imgEl = document.getElementById('c_swap_to_img');
            titleEl.innerText = '标题: 查询中...';
            stockEl.innerText = '库存: 查询中...';
            const info = await fetchSkuInfo(targetSku);
            if (info) {
                titleEl.innerText = info.title;
                stockEl.innerText = `库存: ${info.available}`;
                if (info.image) { imgEl.src = info.image; imgEl.style.display = 'block'; }
                else { imgEl.style.display = 'none'; }
            } else {
                titleEl.innerText = '标题: 未找到';
                stockEl.innerText = '库存: --';
                imgEl.style.display = 'none';
            }
        });

        // 加载推荐SKU列表（按SKU前缀搜索同款，智能排序）
        const recommendEl = document.getElementById('c_swap_recommend');
        const stylePrefix = extractStylePrefix(currentSku);
        recommendEl.innerText = '搜索中...';
        const list = await fetchRelatedSkus(stylePrefix, currentSku);
        if (list.length) {
            const sorted = sortRecommendations(list, currentTitle);
            recommendEl.innerHTML = sorted.map(item =>
                `<div class="c_rec_item" style="display:flex;align-items:center;padding:6px 0;border-bottom:1px solid #f0f0f0;cursor:pointer;gap:8px;" data-sku="${item.sku}">
                    <img src="${item.image}" style="width:40px;height:40px;object-fit:cover;border-radius:4px;flex-shrink:0;${item.image ? '' : 'display:none;'}" />
                    ${item._label ? `<span style="padding:1px 4px;border-radius:3px;font-size:10px;white-space:nowrap;background:${item._priority === 1 ? '#e6f7ff;color:#1890ff' : item._priority === 2 ? '#f6ffed;color:#52c41a' : '#fff7e6;color:#fa8c16'};">${item._label}</span>` : ''}
                    <span style="flex:1;word-break:break-all;"><strong>${item.title}</strong><br/>${item.sku}</span>

                el.addEventListener('click', () => {
                    const sku = el.dataset.sku;
                    toInput.value = sku;
                    toInput.dispatchEvent(new Event('blur'));
                });
            });
        } else {
            recommendEl.innerText = '未找到同款SKU';
        }
    }

    // 在页面SKU下方注入换货结果
    function injectSwapResult(skuTag, targetSku, targetInfo) {
        const skuCell = skuTag.closest('td.merge_el.dt_el');
        if (!skuCell) return;
        // 避免重复注入
        const existing = skuCell.querySelector('.c_swap_result');
        if (existing) existing.remove();

        const resultDiv = document.createElement('div');
        resultDiv.className = 'c_swap_result';
        resultDiv.style.cssText = 'margin-top:8px;padding:8px;border:1px solid #52c41a;border-radius:6px;background:#f6ffed;';
        resultDiv.innerHTML = `
            <div style="text-align:center;color:#52c41a;font-size:14px;">↓</div>
            <div style="display:flex;align-items:center;gap:8px;">
                <img src="${targetInfo.image}" style="width:40px;height:40px;object-fit:cover;border-radius:4px;${targetInfo.image ? '' : 'display:none;'}" />
                <div style="flex:1;">
                    <div style="font-size:13px;font-weight:bold;word-break:break-all;">${targetSku}</div>
                    <div style="font-size:11px;color:#666;word-break:break-all;">${targetInfo.title}</div>
                    <div style="font-size:11px;color:#52c41a;">库存: ${targetInfo.available}</div>
                </div>
            </div>
        `;
        skuCell.appendChild(resultDiv);
    }

    function injectButtons() {
        const tags = document.querySelectorAll('td.merge_el.dt_el div.copy_txt_box > div > span');
        tags.forEach(tag => {
            if (tag.nextElementSibling && tag.nextElementSibling.classList.contains('c_swap_btn')) return;
            const btn = document.createElement('button');
            btn.className = 'c_swap_btn';
            btn.innerText = '换货';
            btn.style.cssText = STYLE;
            btn.onclick = async () => {
                const sku = tag.innerText.trim();
                // 获取同行订单号
                const row = tag.closest('tr');
                const orderSpan = row ? row.querySelector('td.odn_el div.copy_txt_box > div > span') : null;
                const orderNo = orderSpan ? orderSpan.innerText.trim() : null;
                // 通过接口查询当前SKU信息（标题、库存、图片）
                const info = await fetchSkuInfo(sku);
                const stock = info ? String(info.available) : '--';
                const title = info ? info.title : '--';
                const image = info ? info.image : '';
                showSwapModal(sku, stock, title, image, orderNo, tag);
            };
            tag.parentElement.insertBefore(btn, tag.nextSibling);
        });
    }

    // 页面加载时从云端查询换货记录并注入显示
    async function loadSwapRecordsForPage() {
        // 收集当前页所有订单号
        const orderSpans = document.querySelectorAll('td.odn_el div.copy_txt_box > div > span');
        const orderNos = [];
        orderSpans.forEach(span => {
            const no = span.innerText.trim();
            if (no && /^\d+$/.test(no)) orderNos.push(no);
        });
        console.log('[Bigseller助手] 收集到订单号:', orderNos.length, orderNos.slice(0, 5));
        if (!orderNos.length) return;

        const records = await querySwapRecords(orderNos);
        console.log('[Bigseller助手] 云端返回换货记录:', records.length, records);
        if (!records.length) return;

        // 按订单号分组
        const recordMap = {};
        records.forEach(r => {
            if (!recordMap[r.orderNo]) recordMap[r.orderNo] = [];
            recordMap[r.orderNo].push(r);
        });
        console.log('[Bigseller助手] 记录分组:', Object.keys(recordMap));

        // 遍历页面，为有换货记录的SKU注入结果
        const allRows = document.querySelectorAll('tr');
        console.log('[Bigseller助手] 页面tr数量:', allRows.length);
        let matched = 0;
        allRows.forEach(row => {
            const oSpan = row.querySelector('td.odn_el div.copy_txt_box > div > span');
            if (!oSpan) return;
            const orderNo = oSpan.innerText.trim();
            if (!recordMap[orderNo]) return;

            const skuSpan = row.querySelector('td.merge_el.dt_el div.copy_txt_box > div > span');
            if (!skuSpan) return;
            const sku = skuSpan.innerText.trim();

            const match = recordMap[orderNo].find(r => r.originalSku === sku);
            if (!match) {
                console.log('[Bigseller助手] 订单', orderNo, '有记录但SKU不匹配, 页面SKU:', sku, '记录SKU:', recordMap[orderNo].map(r => r.originalSku));
                return;
            }

            const skuCell = skuSpan.closest('td.merge_el.dt_el');
            if (!skuCell || skuCell.querySelector('.c_swap_result')) return;

            matched++;
            const resultDiv = document.createElement('div');
            resultDiv.className = 'c_swap_result';
            resultDiv.style.cssText = 'margin-top:8px;padding:8px;border:1px solid #52c41a;border-radius:6px;background:#f6ffed;';
            resultDiv.innerHTML = `
                <div style="text-align:center;color:#52c41a;font-size:14px;">↓</div>
                <div style="display:flex;align-items:center;gap:8px;">
                    <img src="${match.targetImage}" style="width:40px;height:40px;object-fit:cover;border-radius:4px;${match.targetImage ? '' : 'display:none;'}" />
                    <div style="flex:1;">
                        <div style="font-size:13px;font-weight:bold;word-break:break-all;">${match.targetSku}</div>
                        <div style="font-size:11px;color:#666;word-break:break-all;">${match.targetTitle}</div>
                        <div style="font-size:11px;color:#52c41a;">库存: ${match.targetAvailable}</div>
                    </div>
                </div>
            `;

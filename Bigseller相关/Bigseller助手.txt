    async function fetchWarehouseList() {
        try {
            const resp = await fetch('/api/v1/warehouse/pageList.json', {
                method: 'POST',
                headers: { 'Content-Type': 'application/x-www-form-urlencoded', 'clienttype': '1' },
                body: 'orderBy=&desc='

            const json = await resp.json();
            if (json.code === 0 && json.data?.rows?.length) {
                return json.data.rows
                    .filter(w => w.permission === 1)
                    .map(w => ({ id: String(w.id), name: w.name, isDefault: w.isDefault === 1 }));
            }
        } catch (e) {
            console.log('[Bigseller助手] 获取仓库列表失败:', e.message || e);
        }
        return [];

    const packageWarehouseMap = {}; // packageNo → warehouseName
    const packageOrderMap = {};     // packageNo → platformOrderId

    // 拦截 XMLHttpRequest 获取订单列表中的仓库信息（页面用 axios/XHR）
    const origXhrOpen = XMLHttpRequest.prototype.open;
    const origXhrSend = XMLHttpRequest.prototype.send;
    XMLHttpRequest.prototype.open = function (method, url, ...rest) {
        this._url = url;
        return origXhrOpen.call(this, method, url, ...rest);
    };
    XMLHttpRequest.prototype.send = function (...args) {
        this.addEventListener('load', function () {
            try {
                if (this._url && this._url.includes('/api/v1/order/new/pageList.json')) {
                    const json = JSON.parse(this.responseText);
                    if (json.code === 0 && json.data?.page?.rows?.length) {
                        json.data.page.rows.forEach(order => {
                            const whName = order.shipmentWarehouse
                                || (warehouseList.find(w => w.id === String(order.warehouseId))?.name)
                                || '';
                            if (order.packageNo) {
                                if (whName) packageWarehouseMap[order.packageNo] = whName;
                                if (order.platformOrderId) packageOrderMap[order.packageNo] = order.platformOrderId;
                            }
                        });
                        setTimeout(injectOrderUI, 500);
                    }
                }
            } catch (e) {}
        });
        return origXhrSend.apply(this, args);
    };

    // 在包裹号右边注入仓库标签和换货按钮
    function injectOrderUI() {
        const packageSpans = document.querySelectorAll('div.list_items_out > div > div.lt > div > div > span');
        packageSpans.forEach(span => {
            const text = span.innerText.trim().replace(/^#\s*/, '');
            if (!text || !packageWarehouseMap[text]) return;
            const parent = span.parentElement;
            // 注入仓库标签
            if (!parent.querySelector('.c_wh_tag')) {
                const tag = document.createElement('span');
                tag.className = 'c_wh_tag';
                tag.innerText = packageWarehouseMap[text];
                tag.style.cssText = 'margin-left:8px;padding:1px 6px;background:#f0f5ff;border:1px solid #adc6ff;color:#2f54eb;border-radius:3px;font-size:11px;vertical-align:middle;';
                parent.insertBefore(tag, span.nextSibling);
            }
            // 注入换货按钮
            if (!parent.querySelector('.c_order_swap_btn')) {
                const btn = document.createElement('button');
                btn.className = 'c_order_swap_btn';
                btn.innerText = '换货';
                btn.style.cssText = STYLE;
                btn.onclick = () => {
                    const orderNo = packageOrderMap[text];
                    if (orderNo) showOrderSwapModal(orderNo);
                    else alert('未找到订单号');
                };
                parent.appendChild(btn);
            }
        });
    }

    // GM_xmlhttpRequest 封装为 Promise
    function gmFetch(url, data) {
        return new Promise((resolve, reject) => {
            console.log('[Bigseller助手][gmFetch] 发送请求:', url, JSON.stringify(data).substring(0, 200));
            GM_xmlhttpRequest({
                method: 'POST',
                url,
                headers: { 'Content-Type': 'application/json' },
                data: JSON.stringify(data),
                onload: (resp) => {
                    console.log('[Bigseller助手][gmFetch] 响应状态:', resp.status, '内容:', resp.responseText.substring(0, 500));
                    try { resolve(JSON.parse(resp.responseText)); }
                    catch (e) { console.error('[Bigseller助手][gmFetch] JSON解析失败:', resp.responseText.substring(0, 200)); reject(e); }
                },
                onerror: (e) => { console.error('[Bigseller助手][gmFetch] 网络错误:', e); reject(e); }
            });
        });
    }

    async function fetchSkuInfo(sku, warehouseId) {
        const whId = warehouseId || getWarehouseId();
        const params = new URLSearchParams({
            pageNo: '1', pageSize: '50', searchType: 'skuName', searchContent: sku,
            inquireType: '2', stockStatus: '', isGroup: '', orderBy: '', desc: '',
            fullCid: '', queryDistribution: '1', saleState: '', zoneId: '',
            openFlag: 'false', hideZeroInventorySku: '0', warehouseIds: whId
        });
        const resp = await fetch('/api/v1/inventory/pageList.json', {
            method: 'POST',
            headers: { 'Content-Type': 'application/x-www-form-urlencoded', 'clienttype': '1' },
            body: params.toString()
        });
        const json = await resp.json();
        if (json.code === 0 && json.data?.page?.rows?.length) {
            const exact = json.data.page.rows.find(r => r.sku === sku || r.skuName === sku);
            const item = exact || json.data.page.rows[0];
            return { title: item.title || '--', available: item.available ?? '--', image: item.image || '', skuId: item.skuId || null };
        }
        console.log('[Bigseller助手][fetchSkuInfo] 未找到 sku=%s, wh=%s, rows=%d', sku, whId, json.data?.page?.rows?.length || 0);
        return null;
    }

    // SKU 工具函数
    function extractStylePrefix(sku) {
        const parts = sku.split('-');
        if (parts.length >= 2) return parts.slice(0, 2).join('-');
        return sku;
    }
    function parseTitleInfo(title) {
        const parts = title.split(';');
        const pattern = parts.length >= 2 ? parts[parts.length - 2].trim() : '';
        const sizeMatch = title.match(/(\d+)\s*[\*xX×]\s*(\d+)/);
        const width = sizeMatch ? parseInt(sizeMatch[1]) : 0;
        const height = sizeMatch ? parseInt(sizeMatch[2]) : 0;
        return { pattern, width, height, area: width * height };
    }
    function sortRecommendations(list, currentTitle) {
        const current = parseTitleInfo(currentTitle);
        function getPriority(info) {
            const sameSize = info.width === current.width && info.height === current.height;
            const samePattern = info.pattern === current.pattern;
            const largerSize = info.area > current.area;
            if (sameSize && !samePattern) return { level: 1, label: '异图同尺' };
            if (samePattern && largerSize) return { level: 2, label: '同图更大' };
            if (!samePattern && largerSize) return { level: 3, label: '异图更大' };
            return { level: 4, label: '' };
        }
        return list.map(item => {
            const info = parseTitleInfo(item.title);
            const pri = getPriority(info);
            return { ...item, _priority: pri.level, _label: pri.label, _info: info };
        }).sort((a, b) => {
            const aHas = a.available > 0 ? 1 : 0, bHas = b.available > 0 ? 1 : 0;
            if (aHas !== bHas) return bHas - aHas;
            if (a._priority !== b._priority) return a._priority - b._priority;
            if (a._priority === 2 || a._priority === 3) {
                const aSH = a._info.height === current.height ? 0 : 1;
                const bSH = b._info.height === current.height ? 0 : 1;
                if (aSH !== bSH) return aSH - bSH;
                if (a._info.area !== b._info.area) return a._info.area - b._info.area;
            }
            return (b.available ?? 0) - (a.available ?? 0);
        });
    }

            input.addEventListener('keydown', e => { if (e.key === 'Enter') { e.preventDefault(); queryTarget(); } });
            // 输入时实时模糊筛选推荐列表
            input.addEventListener('input', () => {
                const row = input.closest('.c_sku_row');
                const recList = row.querySelector('.c_rec_list');
                const keyword = input.value.trim().toLowerCase();
                const items = recList.querySelectorAll('.c_rec_item');
                if (!items.length) return;
                recList.style.display = 'block';
                if (!keyword) {
                    items.forEach(el => el.style.display = '');
                } else {
                    // 模糊匹配：SKU或标题包含关键词
                    items.forEach(el => {
                        const sku = (el.dataset.sku || '').toLowerCase();
                        const title = (el.textContent || '').toLowerCase();
                        el.style.display = (sku.includes(keyword) || title.includes(keyword)) ? '' : 'none';
                    });
                }
            });
        });

        // 推荐按钮（切换显示/隐藏）+ 自动加载推荐
        const recCache = {}; // sku → { html, handlers } 缓存推荐列表
        async function loadRecForRow(row) {
            const recList = row.querySelector('.c_rec_list');
            const sku = row.dataset.sku;
            recList.style.display = 'block';
            // 有缓存直接用
            if (recCache[sku]) {
                recList.innerHTML = recCache[sku];
                bindRecItemEvents(row);
                return;
            }
            recList.innerHTML = '搜索中...';
            const prefix = extractStylePrefix(sku);
            const list = await fetchRelatedSkus(prefix, sku);
            if (!list.length) { recList.innerHTML = '未找到同款SKU'; return; }
            const info = await fetchSkuInfo(sku);
            const currentTitle = info ? info.title : '';
            const sorted = sortRecommendations(list, currentTitle);
            const html = sorted.map(item =>
                `<div class="c_rec_item" style="display:flex;align-items:center;padding:4px 0;border-bottom:1px solid #f0f0f0;cursor:pointer;gap:6px;" data-sku="${item.sku}">
                    <img src="${item.image}" style="width:30px;height:30px;object-fit:cover;border-radius:3px;${item.image ? '' : 'display:none;'}" />
                    ${item._label ? `<span style="padding:1px 3px;border-radius:2px;font-size:10px;white-space:nowrap;background:${item._priority === 1 ? '#e6f7ff;color:#1890ff' : item._priority === 2 ? '#f6ffed;color:#52c41a' : '#fff7e6;color:#fa8c16'};">${item._label}</span>` : ''}
                    <span style="flex:1;word-break:break-all;font-size:11px;"><strong>${item.title}</strong><br/>${item.sku}</span>
                    <span style="white-space:nowrap;color:${item.available > 0 ? '#52c41a' : '#999'};font-size:11px;">可用:${item.available}</span>
                </div>`
            ).join('');
            recCache[sku] = html;
            recList.innerHTML = html;
            bindRecItemEvents(row);
        }
        function bindRecItemEvents(row) {
            const recList = row.querySelector('.c_rec_list');
            recList.querySelectorAll('.c_rec_item').forEach(el => {
                el.addEventListener('click', async () => {
                    const input = row.querySelector('.c_target_input');
                    const infoEl = row.querySelector('.c_target_info');
                    input.value = el.dataset.sku;
                    recList.style.display = 'none';
                    infoEl.innerHTML = '<span style="color:#999;">查询中...</span>';
                    const info = await fetchSkuInfo(el.dataset.sku);
                    if (info) {
                        infoEl.innerHTML = `<span style="color:#333;">${info.title}</span> | <span style="color:#52c41a;">库存: ${info.available}</span>`;
                    } else {
                        infoEl.innerHTML = '<span style="color:#ff4d4f;">未找到该SKU</span>';
                    }
                });
            });
        }
        mask.querySelectorAll('.c_rec_btn').forEach(btn => {
            btn.addEventListener('click', async () => {
                const row = btn.closest('.c_sku_row');
                const recList = row.querySelector('.c_rec_list');
                if (recList.style.display !== 'none') { recList.style.display = 'none'; }
                else { await loadRecForRow(row); }
            });
        });
        // 弹窗打开时：如果当前选择的仓库和订单仓库不同，自动填充目标SKU信息
        if (currentWarehouseId && currentWarehouseId !== orderWarehouseId) {
            mask.querySelectorAll('.c_sku_row').forEach(async row => {
                const sku = row.dataset.sku;
                const input = row.querySelector('.c_target_input');
                const infoEl = row.querySelector('.c_target_info');
                input.value = sku;
                infoEl.innerHTML = '<span style="color:#999;">查询中...</span>';
                const info = await fetchSkuInfo(sku);
                if (info) {
                    infoEl.innerHTML = `<span style="color:#333;">${info.title}</span> | <span style="color:#52c41a;">库存: ${info.available}</span>`;
                } else {
                    infoEl.innerHTML = '<span style="color:#ff4d4f;">目标仓库未找到该SKU</span>';
                }
            });
        }
        // 始终加载推荐列表（供输入筛选用）
        mask.querySelectorAll('.c_sku_row').forEach(row => loadRecForRow(row));

        // 加载每个SKU的同SKU订单数量
        const sameSkuOrdersMap = {}; // sku → orders[]
        mask.querySelectorAll('.c_sku_row').forEach(async row => {
            const sku = row.dataset.sku;
            const orders = await fetchSameSkuOrders(sku, orderNo);
            const batchArea = row.querySelector('.c_batch_area');
            batchArea.style.display = 'block';
            if (orders.length > 0) {
                sameSkuOrdersMap[sku] = orders;
                row.querySelector('.c_batch_count').innerText = orders.length;
            } else {
                batchArea.innerHTML = '<span style="font-size:11px;color:#999;">没有其他同SKU待处理订单</span>';
            }
        });

        // 确认换货
        document.getElementById('c_swap_confirm').onclick = async () => {
            const rows = mask.querySelectorAll('.c_sku_row');
            const tasks = [];
            rows.forEach(row => {
                const targetSku = row.querySelector('.c_target_input').value.trim();
                const originalSku = row.dataset.sku;
                const rootSku = row.dataset.rootSku || originalSku;
                const batchChecked = row.querySelector('.c_batch_check')?.checked || false;
                // 有目标SKU就加入任务；没有目标SKU但换了仓库也加入（纯换仓）
                if (targetSku) {
                    tasks.push({ originalSku, rootSku, targetSku, orderItemId: row.dataset.orderItemId, batchChecked });
                } else if (getWarehouseId() !== orderWarehouseId) {
                    tasks.push({ originalSku, rootSku, targetSku: originalSku, orderItemId: row.dataset.orderItemId, batchChecked });
                }
            });
            if (!tasks.length) { alert('请至少为一个SKU指定目标'); return; }

            const confirmBtn = document.getElementById('c_swap_confirm');
            confirmBtn.innerText = '换货中...';
            confirmBtn.disabled = true;
            let success = 0, fail = 0;
            const successItems = []; // 收集成功的换货项，最后统一保存

            for (let i = 0; i < tasks.length; i++) {
                const t = tasks[i];
                confirmBtn.innerText = `换货中 ${i + 1}/${tasks.length}...`;
                console.log('[Bigseller助手][换货][DEBUG] 任务%d:', i, JSON.stringify({ orderNo, orderId, originalSku: t.originalSku, rootSku: t.rootSku, targetSku: t.targetSku, orderItemId: t.orderItemId }));
                try {
                    const selectedWhId = getWarehouseId();
                    const isSameSku = t.targetSku === t.originalSku;
                    const isSameWarehouse = selectedWhId === orderWarehouseId;
                    console.log('[Bigseller助手][换货][DEBUG] selectedWhId=%s, orderWarehouseId=%s, isSameSku=%s, isSameWarehouse=%s', selectedWhId, orderWarehouseId, isSameSku, isSameWarehouse);

                    if (isSameSku && isSameWarehouse) {
                        console.log('[Bigseller助手][换货] 同仓同SKU，跳过'); continue;
                    }

                    let swapOk = false;

                    // 步骤1：如果需要换仓，先调 changeMappingWarehouse
                    if (!isSameWarehouse) {
                        console.log('[Bigseller助手][换货] 步骤1: 换仓 %s → %s', orderWarehouseId, selectedWhId);
                        const whOk = await doChangeWarehouse(orderId, selectedWhId);
                        if (!whOk) { console.log('[Bigseller助手][换货] 换仓失败'); fail++; continue; }
                        console.log('[Bigseller助手][换货] 换仓成功');
                    }

                    // 步骤2：如果需要换SKU，调 changeMappingRelation
                    if (!isSameSku) {
                        const targetInfo = await fetchSkuInfo(t.targetSku);
                        if (!targetInfo || !targetInfo.skuId) { console.log('[Bigseller助手][换货][DEBUG] 目标SKU未找到:', t.targetSku); fail++; continue; }
                        console.log('[Bigseller助手][换货] 步骤2: 换SKU %s → %s (skuId=%s)', t.originalSku, t.targetSku, targetInfo.skuId);
                        swapOk = await doSwapCall(orderId, t.orderItemId, selectedWhId, targetInfo.skuId, t.targetSku);
                        console.log('[Bigseller助手][换货][DEBUG] doSwapCall结果: swapOk=%s', swapOk);
                        if (swapOk) {
                            successItems.push({ rootSku: t.rootSku, originalSku: t.originalSku, targetSku: t.targetSku, targetSkuId: String(targetInfo.skuId), targetTitle: targetInfo.title, targetImage: targetInfo.image, targetAvailable: targetInfo.available });
                            const skuSpan = findSkuSpanInOrder(orderNo, t.rootSku) || findSkuSpanInOrder(orderNo, t.originalSku);
                            if (skuSpan) injectSwapResult(skuSpan, t.targetSku, targetInfo, t.rootSku, orderWarehouseId);
                        }
                    } else {
                        // 纯换仓（不换SKU），换仓已经在步骤1完成
                        swapOk = true;
                        const whName = warehouseList.find(w => w.id === selectedWhId)?.name || selectedWhId;
                        const origInfo = await fetchSkuInfo(t.originalSku);
                        successItems.push({ rootSku: t.rootSku, originalSku: t.originalSku, targetSku: t.originalSku, targetSkuId: origInfo?.skuId ? String(origInfo.skuId) : '', targetTitle: origInfo?.title || '', targetImage: origInfo?.image || '', targetAvailable: origInfo?.available || '' });
                        const skuSpan = findSkuSpanInOrder(orderNo, t.rootSku) || findSkuSpanInOrder(orderNo, t.originalSku);
                        if (skuSpan) {
                            injectSwapResult(skuSpan, t.originalSku, {
                                title: `${origInfo?.title || ''} [${whName}]`,
                                available: origInfo?.available ?? '--',
                                image: origInfo?.image || ''
                            }, t.rootSku, orderWarehouseId);
                        }
                    }

                    if (swapOk) {
                        success++;
                        // 批量应用到其他同SKU订单
                        if (t.batchChecked && sameSkuOrdersMap[t.originalSku]?.length) {
                            const batchOrders = sameSkuOrdersMap[t.originalSku];
                            const skippedMultiSku = []; // 记录跳过的多SKU订单
                            for (let j = 0; j < batchOrders.length; j++) {
                                try {
                                    const bo = batchOrders[j];
                                    const dResp = await fetch(`/api/v1/order/detail.json?id=${bo.id}&historyOrder=0&viewBuyerMessage=false`, { headers: { 'clienttype': '1' } });
                                    const dJson = await dResp.json();
                                    if (dJson.code !== 0 || !dJson.data?.orderDetail?.orderItemVoList?.length) continue;
                                    const itemList = dJson.data.orderDetail.orderItemVoList;
                                    // 多SKU订单跳过，避免换仓导致其他SKU异仓
                                    if (itemList.length > 1 && selectedWhId !== String(bo.warehouseId || orderWarehouseId)) {
                                        skippedMultiSku.push(bo.platformOrderId);
                                        continue;
                                    }
                                    const oItem = itemList.find(item => item.varSku === t.originalSku || item.inventorySku === t.originalSku);
                                    if (!oItem) continue;
                                    const boWhId = String(bo.warehouseId || orderWarehouseId);
                                    let bOk = false;
                                    if (selectedWhId !== boWhId) {
                                        const bWhOk = await doChangeWarehouse(bo.id, selectedWhId);
                                        if (!bWhOk) continue;
                                    }
                                    if (!isSameSku) {
                                        const bTargetInfo = await fetchSkuInfo(t.targetSku);
                                        if (bTargetInfo?.skuId) {
                                            bOk = await doSwapCall(bo.id, oItem.id, selectedWhId, bTargetInfo.skuId, t.targetSku);
                                        }
                                    } else {
                                        bOk = true;
                                    }
                                    if (bOk) {
                                        const bItem = { rootSku: oItem.varSku || t.originalSku, originalSku: t.originalSku, targetSku: isSameSku ? t.originalSku : t.targetSku, targetSkuId: '', targetTitle: '', targetImage: '', targetAvailable: '' };
                                        saveSwapRecord({ orderNo: bo.platformOrderId, orderId: String(bo.id), rootWarehouseId: boWhId, items: [bItem] });
                                    }
                                } catch (e) {}
                                confirmBtn.innerText = `批量 ${t.originalSku} ${j + 1}/${batchOrders.length}...`;
                            }
                            if (skippedMultiSku.length) {
                                alert(`以下多SKU订单已跳过（需换仓但可能导致其他SKU异仓），请手动处理：\n${skippedMultiSku.join('\n')}`);
                            }
                        }
                    }
                } catch (e) { fail++; console.log('[Bigseller助手][换货] 异常:', e.message); }
            }

            // 统一保存本订单的换货记录
            if (successItems.length) {
                const saveResult = await saveSwapRecord({ orderNo, orderId: String(orderId), rootWarehouseId: orderWarehouseId, items: successItems });
                console.log('[Bigseller助手][换货][DEBUG] 统一保存结果:', JSON.stringify(saveResult));
            }

            alert(`换货完成！成功: ${success}，失败: ${fail}`);
            // 刷新仓库标签显示
            if (success > 0) {
                const selectedWhId = getWarehouseId();
                const whName = warehouseList.find(w => w.id === selectedWhId)?.name || selectedWhId;
                // 找到当前订单的包裹号并更新 packageWarehouseMap
                Object.keys(packageOrderMap).forEach(pkgNo => {
                    if (packageOrderMap[pkgNo] === orderNo) {
                        packageWarehouseMap[pkgNo] = whName;
                        // 更新页面上的仓库标签
                        const tag = document.querySelector(`.c_wh_tag`);
                        document.querySelectorAll('.c_wh_tag').forEach(el => {
                            const span = el.previousSibling;
                            if (span && span.innerText && span.innerText.trim().replace(/^#\s*/, '') === pkgNo) {
                                el.innerText = whName;
                            }
                        });
                    }
                });
            }
            closeModal();
        };
    }

    // 取消换货核心逻辑
    async function doCancelSwap(orderNo, originalSku, targetSku, rootSku, rootWarehouseId) {
        try {
            console.log('[Bigseller助手][取消换货][DEBUG] 开始:', JSON.stringify({ orderNo, originalSku, targetSku, rootSku, rootWarehouseId }));
            // 查询订单当前状态
            const orderResp = await fetch('/api/v1/order/new/pageList.json', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json', 'clienttype': '1' },
                body: JSON.stringify({ searchType: 'orderNo', searchContent: orderNo, inquireType: 2, allOrder: true, historyOrder: 0, pageNo: 1, pageSize: 10, desc: 1, orderBy: 'printTime' })
            });
            const orderJson = await orderResp.json();
            if (orderJson.code !== 0 || !orderJson.data?.page?.rows?.length) return { success: false, msg: '未找到订单' };
            const cancelOrderRow = orderJson.data.page.rows[0];
            const orderId = cancelOrderRow.id;
            const currentWhId = String(cancelOrderRow.warehouseId);
            const needChangeWh = rootWarehouseId && rootWarehouseId !== currentWhId;
            const needChangeSku = rootSku !== targetSku;
            console.log('[Bigseller助手][取消换货][DEBUG] orderId=%s, currentWhId=%s, rootWarehouseId=%s, needChangeWh=%s, needChangeSku=%s, 恢复目标: rootSku=%s', orderId, currentWhId, rootWarehouseId, needChangeWh, needChangeSku, rootSku);

            // 步骤1：如果需要换仓，先换回原仓库
            if (needChangeWh) {
                const whOk = await doChangeWarehouse(orderId, rootWarehouseId);
                if (!whOk) return { success: false, msg: '换回原仓库失败' };
                console.log('[Bigseller助手][取消换货][DEBUG] 换仓成功 → %s', rootWarehouseId);
            }

            // 步骤2：如果需要换SKU，换回rootSku（平台原始SKU）
            if (needChangeSku) {
                const restoreWhId = rootWarehouseId || currentWhId;
                // 在恢复目标仓库中查rootSku信息
                const rootSkuInfo = await fetchSkuInfo(rootSku, restoreWhId);
                console.log('[Bigseller助手][取消换货][DEBUG] fetchSkuInfo(%s, wh=%s) 结果:', rootSku, restoreWhId, rootSkuInfo ? JSON.stringify({ skuId: rootSkuInfo.skuId }) : 'null');
                if (!rootSkuInfo || !rootSkuInfo.skuId) return { success: false, msg: `未找到根SKU信息: ${rootSku} (仓库: ${restoreWhId})` };
                // 获取订单详情找到对应orderItemId
                const detailResp = await fetch(`/api/v1/order/detail.json?id=${orderId}&historyOrder=0&viewBuyerMessage=false`, { headers: { 'clienttype': '1' } });
                const detailJson = await detailResp.json();
                if (detailJson.code !== 0 || !detailJson.data?.orderDetail?.orderItemVoList?.length) return { success: false, msg: '获取订单详情失败' };
                const allItems = detailJson.data.orderDetail.orderItemVoList;
                // 用rootSku(varSku)匹配商品项
                const orderItem = allItems.find(item => item.varSku === rootSku);
                if (!orderItem) return { success: false, msg: `未找到商品项: ${rootSku}` };
                const swapOk = await doSwapCall(orderId, orderItem.id, restoreWhId, rootSkuInfo.skuId, rootSku);
                if (!swapOk) return { success: false, msg: '换回原SKU失败' };
                console.log('[Bigseller助手][取消换货][DEBUG] 换SKU成功 → %s', rootSku);
            }

            return { success: true, msg: '成功' };
        } catch (e) { return { success: false, msg: e.message }; }
    }

    async function cancelSwap(orderNo, originalSku, targetSku, rootSku, resultDiv) {
        if (!confirm('确认取消换货？将恢复为原SKU映射')) return;
        const result = await doCancelSwap(orderNo, originalSku, targetSku, rootSku);
        if (result.success) { alert('已取消换货，恢复为原SKU'); resultDiv.remove(); }
        else { alert('取消换货失败: ' + result.msg); }
    }

    // 订单级取消换货：取消该订单所有换货记录
    async function cancelOrderSwap(orderNo) {
        const records = await querySwapRecords([orderNo]);
        if (!records.length) { alert('该订单没有换货记录'); return; }
        const record = records[0]; // 新结构：一订单一条记录
        const items = record.items || [];
        if (!items.length) { alert('该订单没有换货项'); return; }
        if (!confirm(`确认取消该订单的全部 ${items.length} 条换货？将恢复原SKU映射`)) return;
        let success = 0; const failures = [];
        for (const item of items) {
            const result = await doCancelSwap(orderNo, item.originalSku, item.targetSku, item.rootSku, record.rootWarehouseId);
            if (result.success) success++;
            else failures.push(`${item.rootSku}: ${result.msg}`);
        }
        // 成功后删除整条记录
        if (success > 0) {
            await deleteSwapRecord(orderNo);
        }
        // 移除页面上该订单的换货样式，恢复原SKU显示
        const odnSpans = document.querySelectorAll('td.odn_el div.copy_txt_box > div > span');
        for (const sp of odnSpans) {
            if (sp.innerText.trim() === orderNo) {
                const row = sp.closest('tr');
                if (!row) continue;
                row.querySelectorAll('.c_swap_styled').forEach(container => {
                    // 恢复SKU文本
                    const skuSpan = container.querySelector('span.f_default');
                    if (skuSpan && skuSpan.dataset.rootSku) {
                        skuSpan.innerText = skuSpan.dataset.rootSku;
                        delete skuSpan.dataset.rootSku;
                    }
                    // 恢复商品图片
                    const img = container.querySelector('.img_box_out img');
                    if (img && img.dataset.rootImage) {
                        img.src = img.dataset.rootImage;
                        delete img.dataset.rootImage;
                    }
                    // 恢复变体描述
                    const preInit = container.querySelector('span.pre_init');
                    if (preInit && preInit.dataset.rootTitle) {
                        preInit.innerText = preInit.dataset.rootTitle;
                        delete preInit.dataset.rootTitle;
                    }
                    // 移除标签和?图标
                    container.querySelectorAll('.c_swap_tag, .c_swap_info').forEach(el => el.remove());
                    // 恢复复制按钮
                    const copyBtn = container.querySelector('.copy_txt_box a.copy_btn');
                    if (copyBtn) copyBtn.style.display = '';
                    // 移除容器样式
                    container.classList.remove('c_swap_styled');
                    container.style.background = '';
                    container.style.border = '';
                    container.style.borderRadius = '';
                    container.style.padding = '';
                });
                // 移除"取消换货"按钮
                const pkgNo = Object.keys(packageOrderMap).find(k => packageOrderMap[k] === orderNo);
                if (pkgNo) {
                    document.querySelectorAll('.c_order_cancel_swap_btn').forEach(btn => {
                        const span = btn.parentElement?.querySelector('span');
                        if (span && span.innerText.trim().replace(/^#\s*/, '') === pkgNo) btn.remove();
                    });
                }
            }
        }
        // 更新仓库标签（恢复到rootWarehouseId对应的仓库名）
        if (success > 0 && record.rootWarehouseId) {
            const rootWhId = record.rootWarehouseId;
            const rootWhName = warehouseList.find(w => w.id === rootWhId)?.name || rootWhId;
            Object.keys(packageOrderMap).forEach(pkgNo => {
                if (packageOrderMap[pkgNo] === orderNo) {
                    packageWarehouseMap[pkgNo] = rootWhName;
                    document.querySelectorAll('.c_wh_tag').forEach(el => {
                        const span = el.previousSibling;
                        if (span && span.innerText && span.innerText.trim().replace(/^#\s*/, '') === pkgNo) {
                            el.innerText = rootWhName;
                        }
                    });
                }
            });
        }
        if (failures.length) {
            alert(`取消完成！成功: ${success}，失败: ${failures.length}\n\n失败详情:\n${failures.join('\n')}`);
        } else {
            alert(`取消完成！全部 ${success} 条成功`);
        }
    }

    // 在页面SKU下方注入换货结果
    // 在订单内按SKU文本精确找到对应的skuSpan（解决多SKU在同一个td内的问题）
    function findSkuSpanInOrder(orderNo, sku) {
        const odnSpans = document.querySelectorAll('td.odn_el div.copy_txt_box > div > span');
        for (const odSpan of odnSpans) {
            if (odSpan.innerText.trim() !== orderNo) continue;
            const row = odSpan.closest('tr');
            if (!row) continue;
            // 多SKU订单：所有SKU都在同一个 td.merge_el.dt_el 中，作为多个 copy_txt_box
            const skuSpans = row.querySelectorAll('td.merge_el.dt_el div.copy_txt_box > div > span.f_default');
            for (const sp of skuSpans) {
                // 匹配当前文本或原始SKU（换货后文本已被替换）
                if (sp.innerText.trim() === sku || sp.dataset.rootSku === sku) return sp;
            }
            console.log('[Bigseller助手][DEBUG] findSkuSpanInOrder 未找到 sku=%s, 订单内所有SKU:', sku, Array.from(skuSpans).map(s => s.innerText.trim()));
        }
        return null;
    }

    function injectSwapResult(skuTag, targetSku, targetInfo, rootSku, rootWarehouseId) {
        // 找到该SKU所属的容器div（td > div > div 级别，每个SKU独占一个）
        const skuCell = skuTag.closest('td.merge_el.dt_el');
        if (!skuCell) return;
        const topDiv = skuCell.firstElementChild;
        const skuContainer = topDiv ? Array.from(topDiv.children).find(child => child.contains(skuTag)) : null;
        const container = skuContainer || skuCell;
        if (container.classList.contains('c_swap_styled')) return; // 已标记过
        const skuSpan = container.querySelector('span.f_default');
        if (!skuSpan) return;
        const originalSku = rootSku || skuSpan.innerText.trim();
        // 替换SKU文本
        skuSpan.innerText = targetSku;
        skuSpan.dataset.rootSku = originalSku;
        // 替换商品图片
        if (targetInfo.image) {
            const img = container.querySelector('.img_box_out img');
            if (img) {
                img.dataset.rootImage = img.src;
                img.src = targetInfo.image;
            }
        }
        // 替换变体描述（span.pre_init）
        if (targetInfo.title) {
            const preInit = container.querySelector('span.pre_init');
            if (preInit) {
                preInit.dataset.rootTitle = preInit.innerText;
                preInit.innerText = targetInfo.title;
            }
        }
        // 容器加样式
        container.classList.add('c_swap_styled');
        container.style.cssText += ';background:#f6ffed;border:1px solid #b7eb8f;border-radius:6px;padding:4px;';
        // 插入"已换货"标签和"?"悬浮
        const copyBox = skuSpan.closest('.copy_txt_box');
        const pointer = copyBox ? copyBox.querySelector('div.pointer') : skuSpan.parentElement;
        // 隐藏原生复制按钮
        if (copyBox) {
            const copyBtn = copyBox.querySelector('a.copy_btn');
            if (copyBtn) copyBtn.style.display = 'none';
        }
        if (!pointer.querySelector('.c_swap_tag')) {
            const tag = document.createElement('span');
            tag.className = 'c_swap_tag';
            tag.style.cssText = 'margin-left:4px;padding:0 7px;background:#f6ffed;border:1px solid #52c41a;color:#52c41a;border-radius:2px;font-size:12px;line-height:20px;display:inline-block;vertical-align:middle;';
            tag.innerText = '已换货';
            const info = document.createElement('span');
            info.className = 'c_swap_info';
            info.style.cssText = 'margin-left:4px;cursor:help;color:#999;font-size:14px;font-weight:bold;vertical-align:middle;padding:0 4px;position:relative;';
            info.innerText = '?';
            info.dataset.rootSku = originalSku;
            info.dataset.rootWhId = rootWarehouseId || '';
            info.addEventListener('mouseenter', showRootSkuPopup);
            info.addEventListener('mouseleave', hideRootSkuPopup);
            pointer.appendChild(info);
            pointer.appendChild(tag);
        }
    }

    // "?"悬浮弹出原SKU信息卡片
    async function showRootSkuPopup(e) {
        const el = e.currentTarget;
        let popup = el.querySelector('.c_root_popup');
        if (popup) { popup.style.display = 'block'; return; }
        popup = document.createElement('div');
        popup.className = 'c_root_popup';
        popup.style.cssText = 'position:absolute;left:0;top:100%;z-index:999999;background:#fff;border:1px solid #d9d9d9;border-radius:6px;box-shadow:0 2px 8px rgba(0,0,0,.15);padding:10px;width:260px;margin-top:4px;';
        popup.innerHTML = '<div style="color:#999;font-size:12px;">加载中...</div>';
        el.appendChild(popup);
        // 从API获取根SKU在根仓库的信息
        const rootSku = el.dataset.rootSku;
        const rootWhId = el.dataset.rootWhId;
        const info = await fetchSkuInfo(rootSku, rootWhId || undefined);
        const img = info?.image ? `<img src="${info.image}" style="width:50px;height:50px;object-fit:cover;border-radius:4px;flex-shrink:0;">` : '';
        popup.innerHTML = `<div style="display:flex;align-items:center;gap:8px;">
            ${img}
            <div style="flex:1;min-width:0;">
                <div style="font-size:13px;font-weight:bold;word-break:break-all;">${rootSku}</div>
                <div style="font-size:11px;color:#666;word-break:break-all;margin-top:2px;">${info?.title || ''}</div>
                <div style="font-size:11px;color:#52c41a;margin-top:2px;">可用库存: ${info?.available ?? '--'}</div>
            </div>
        </div>`;
    }
    function hideRootSkuPopup(e) {
        const popup = e.currentTarget.querySelector('.c_root_popup');
        if (popup) popup.style.display = 'none';
    }

    // 批量取消换货
    function injectBatchCancelBtn() {
        if (document.querySelector('.c_batch_cancel_swap_btn')) return;
        const refBtn = document.querySelector('#CeilingBox > div.rows.flex.star_align > div.lt > div > div > span:nth-child(3) > button');
        if (!refBtn) return;
        const btn = document.createElement('button');
        btn.className = 'c_batch_cancel_swap_btn';
        btn.innerText = '批量取消换货';
        btn.style.cssText = 'margin-left:10px;padding:5px 12px;border:1px solid #ff4d4f;background:#fff;color:#ff4d4f;border-radius:4px;cursor:pointer;font-size:13px;white-space:nowrap;';
        btn.onclick = batchCancelSwap;
        refBtn.parentElement.after(btn);
    }

    async function batchCancelSwap() {
        const rowCheckboxes = document.querySelectorAll('tr td input[type=checkbox]:checked');
        const checkedOrderNos = [];
        rowCheckboxes.forEach(cb => {
            const row = cb.closest('tr');
            if (!row) return;
            const oSpan = row.querySelector('td.odn_el div.copy_txt_box > div > span');
            if (oSpan) { const no = oSpan.innerText.trim(); if (no && /^\d+$/.test(no)) checkedOrderNos.push(no); }
        });
        if (!checkedOrderNos.length) { alert('未勾选任何订单'); return; }
        const records = await querySwapRecords(checkedOrderNos);
        if (!records.length) { alert('勾选的订单中没有换货记录'); return; }
        // 新结构：每条记录有 items 数组，展开为逐项处理
        const allTasks = [];
        records.forEach(r => { (r.items || []).forEach(item => allTasks.push({ orderNo: r.orderNo, rootWarehouseId: r.rootWarehouseId, ...item })); });
        if (!allTasks.length) { alert('没有换货项'); return; }
        if (!confirm(`确认批量取消换货？将对 ${allTasks.length} 条换货恢复原SKU映射`)) return;
        let success = 0, fail = 0;
        const btn = document.querySelector('.c_batch_cancel_swap_btn');
        const originalText = btn.innerText;
        for (let i = 0; i < allTasks.length; i++) {
            const t = allTasks[i];
            btn.innerText = `批量取消中 ${i + 1}/${allTasks.length}...`; btn.disabled = true;
            const result = await doCancelSwap(t.orderNo, t.originalSku, t.targetSku, t.rootSku, t.rootWarehouseId);
            if (result.success) { success++; }
            else { fail++; }
        }
        // 删除所有相关记录并恢复原SKU显示
        const orderNosSet = new Set(records.map(r => r.orderNo));
        for (const oNo of orderNosSet) { await deleteSwapRecord(oNo); }
        document.querySelectorAll('tr').forEach(row => {
            const oSpan = row.querySelector('td.odn_el div.copy_txt_box > div > span');
            if (oSpan && orderNosSet.has(oSpan.innerText.trim())) {
                row.querySelectorAll('.c_swap_styled').forEach(container => {
                    const skuSpan = container.querySelector('span.f_default');
                    if (skuSpan && skuSpan.dataset.rootSku) {
                        skuSpan.innerText = skuSpan.dataset.rootSku;
                        delete skuSpan.dataset.rootSku;
                    }
                    const img = container.querySelector('.img_box_out img');
                    if (img && img.dataset.rootImage) {
                        img.src = img.dataset.rootImage;
                        delete img.dataset.rootImage;
                    }
                    const preInit = container.querySelector('span.pre_init');
                    if (preInit && preInit.dataset.rootTitle) {
                        preInit.innerText = preInit.dataset.rootTitle;
                        delete preInit.dataset.rootTitle;
                    }
                    container.querySelectorAll('.c_swap_tag, .c_swap_info').forEach(el => el.remove());
                    container.classList.remove('c_swap_styled');
                    container.style.background = '';
                    container.style.border = '';
                    container.style.borderRadius = '';
                    container.style.padding = '';
                });
            }
        });
        btn.innerText = originalText; btn.disabled = false;
        alert(`批量取消完成！成功: ${success}，失败: ${fail}`);
    }

    // 页面加载时从云端查询换货记录并注入显示
    let _swapRetryCount = 0;
    async function loadSwapRecordsForPage() {
        const orderSpans = document.querySelectorAll('td.odn_el div.copy_txt_box > div > span');
        const orderNos = [];
        orderSpans.forEach(span => { const no = span.innerText.trim(); if (no && /^\d+$/.test(no)) orderNos.push(no); });
        if (!orderNos.length) return;
        const records = await querySwapRecords(orderNos);
        if (!records.length) return;
        let matched = 0;
        // 新结构：每条记录有 items 数组
        records.forEach(record => {
            const items = record.items || [];
            items.forEach(item => {
                const skuSpan = findSkuSpanInOrder(record.orderNo, item.rootSku);
                if (!skuSpan) return;
                const skuCell = skuSpan.closest('td.merge_el.dt_el');
                if (!skuCell) return;
                const topDiv = skuCell.firstElementChild;
                const skuContainer = topDiv ? Array.from(topDiv.children).find(child => child.contains(skuSpan)) : null;
                const container = skuContainer || skuCell;
                if (container.classList.contains('c_swap_styled')) return;
                matched++;
                // 原位替换SKU文本
                const skuTextSpan = container.querySelector('span.f_default');
                if (!skuTextSpan) return;
                skuTextSpan.innerText = item.targetSku;
                skuTextSpan.dataset.rootSku = item.rootSku;
                // 替换商品图片
                if (item.targetImage) {
                    const img = container.querySelector('.img_box_out img');
                    if (img) {
                        img.dataset.rootImage = img.src;
                        img.src = item.targetImage;
                    }
                }
                // 替换变体描述
                if (item.targetTitle) {
                    const preInit = container.querySelector('span.pre_init');
                    if (preInit) {
                        preInit.dataset.rootTitle = preInit.innerText;
                        preInit.innerText = item.targetTitle;
                    }
                }
                // 容器加样式
                container.classList.add('c_swap_styled');
                container.style.cssText += ';background:#f6ffed;border:1px solid #b7eb8f;border-radius:6px;padding:4px;';
                // 插入"已换货"标签和"?"悬浮
                const copyBox = skuTextSpan.closest('.copy_txt_box');
                const pointer = copyBox ? copyBox.querySelector('div.pointer') : skuTextSpan.parentElement;
                // 隐藏原生复制按钮
                if (copyBox) {
                    const copyBtn = copyBox.querySelector('a.copy_btn');
                    if (copyBtn) copyBtn.style.display = 'none';
                }
                if (!pointer.querySelector('.c_swap_tag')) {
                    const tag = document.createElement('span');
                    tag.className = 'c_swap_tag';
                    tag.style.cssText = 'margin-left:4px;padding:0 7px;background:#f6ffed;border:1px solid #52c41a;color:#52c41a;border-radius:2px;font-size:12px;line-height:20px;display:inline-block;vertical-align:middle;';
                    tag.innerText = '已换货';
                    const info = document.createElement('span');
                    info.className = 'c_swap_info';
                    info.style.cssText = 'margin-left:4px;cursor:help;color:#999;font-size:14px;font-weight:bold;vertical-align:middle;padding:0 4px;position:relative;';
                    info.innerText = '?';
                    info.dataset.rootSku = item.rootSku;
                    info.dataset.rootWhId = record.rootWarehouseId || '';
                    info.addEventListener('mouseenter', showRootSkuPopup);
                    info.addEventListener('mouseleave', hideRootSkuPopup);
                    pointer.appendChild(info);
                    pointer.appendChild(tag);
                }
            });
        });
        // 为有换货记录的订单注入"取消换货"按钮
        records.forEach(record => {
            const orderNo = record.orderNo;
            const pkgNo = Object.keys(packageOrderMap).find(k => packageOrderMap[k] === orderNo);
            if (!pkgNo) return;
            const packageSpans = document.querySelectorAll('div.list_items_out > div > div.lt > div > div > span');
            for (const span of packageSpans) {
                if (span.innerText.trim().replace(/^#\s*/, '') === pkgNo) {
                    const parent = span.parentElement;
                    if (parent && !parent.querySelector('.c_order_cancel_swap_btn')) {
                        const cancelBtn = document.createElement('button');
                        cancelBtn.className = 'c_order_cancel_swap_btn';
                        cancelBtn.innerText = '取消换货';
                        cancelBtn.style.cssText = 'margin-left:6px;padding:2px 8px;border:1px solid #ff4d4f;background:#fff;color:#ff4d4f;border-radius:3px;cursor:pointer;font-size:12px;white-space:nowrap;';
                        cancelBtn.onclick = () => cancelOrderSwap(orderNo);
                        parent.appendChild(cancelBtn);
                    }
                    break;
                }
            }
        });
        if (matched === 0 && records.length > 0 && _swapRetryCount < 5) {
            _swapRetryCount++;
            setTimeout(loadSwapRecordsForPage, _swapRetryCount * 1000);
        } else if (matched > 0) { _swapRetryCount = 0; }
    }

    // 初始化
    async function init() {
        warehouseList = await fetchWarehouseList();
        if (warehouseList.length) {
            const savedId = localStorage.getItem(WH_STORAGE_KEY);
            const validSaved = warehouseList.find(w => w.id === savedId);
            if (validSaved) { currentWarehouseId = savedId; }
            else { const dw = warehouseList.find(w => w.isDefault) || warehouseList[0]; currentWarehouseId = dw.id; localStorage.setItem(WH_STORAGE_KEY, currentWarehouseId); }
        }
    }
    init();

    // 监听DOM变化注入批量取消按钮
    const observer = new MutationObserver(() => injectBatchCancelBtn());
    observer.observe(document.body, { childList: true, subtree: true });
    setTimeout(injectBatchCancelBtn, 1500);

    // 等待订单列表渲染后查询云端换货记录
    let swapDebounceTimer = null;
    let lastOrderListHtml = '';
    const swapObserver = new MutationObserver(() => {
        const orderSpans = document.querySelectorAll('td.odn_el div.copy_txt_box > div > span');
        if (orderSpans.length > 0) {
            // 用订单号列表的指纹判断是否有变化（tab切换、翻页等）
            const currentFingerprint = Array.from(orderSpans).map(s => s.innerText.trim()).join(',');
            if (currentFingerprint !== lastOrderListHtml) {
                lastOrderListHtml = currentFingerprint;
                clearTimeout(swapDebounceTimer);
                swapDebounceTimer = setTimeout(() => {
                    loadSwapRecordsForPage();
                    injectOrderUI();
                }, 800);
            }
        }
    });
    swapObserver.observe(document.body, { childList: true, subtree: true });
})();

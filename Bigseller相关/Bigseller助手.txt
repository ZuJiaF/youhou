    let currentWarehouseId = localStorage.getItem(WH_STORAGE_KEY) || '';

    function getWarehouseId() {
        return currentWarehouseId;
    }

    async function fetchWarehouseList() {
        try {
            const resp = await fetch('/api/v1/warehouse/pageList.json', {
                method: 'POST',
                headers: { 'Content-Type': 'application/x-www-form-urlencoded', 'clienttype': '1' },
                body: 'orderBy=&desc='
            });
            const json = await resp.json();
            if (json.code === 0 && json.data?.rows?.length) {
                return json.data.rows
                    .filter(w => w.permission === 1)
                    .map(w => ({ id: String(w.id), name: w.name, isDefault: w.isDefault === 1 }));
            }
        } catch (e) {
            console.log('[Bigseller助手] 获取仓库列表失败:', e.message || e);
        }
        return [];
    }

    // 订单仓库信息映射（packageNo → warehouseName）
    const packageWarehouseMap = {};

    // 拦截 XMLHttpRequest 获取订单列表中的仓库信息（页面用 axios/XHR）
    const origXhrOpen = XMLHttpRequest.prototype.open;
    const origXhrSend = XMLHttpRequest.prototype.send;
    XMLHttpRequest.prototype.open = function (method, url, ...rest) {
        this._url = url;
        return origXhrOpen.call(this, method, url, ...rest);
    };
    XMLHttpRequest.prototype.send = function (...args) {
        this.addEventListener('load', function () {
            try {
                if (this._url && this._url.includes('/api/v1/order/new/pageList.json')) {
                    const json = JSON.parse(this.responseText);
                    if (json.code === 0 && json.data?.page?.rows?.length) {
                        json.data.page.rows.forEach(order => {
                            const whName = order.shipmentWarehouse
                                || (warehouseList.find(w => w.id === String(order.warehouseId))?.name)
                                || '';
                            if (whName && order.packageNo) {
                                packageWarehouseMap[order.packageNo] = whName;
                            }
                        });
                        setTimeout(injectWarehouseTags, 500);
                    }
                }
            } catch (e) {}
        });
        return origXhrSend.apply(this, args);
    };

    // 在包裹号右边注入仓库标签
    function injectWarehouseTags() {
        const packageSpans = document.querySelectorAll('div.list_items_out > div > div.lt > div > div > span');
        packageSpans.forEach(span => {
            const text = span.innerText.trim().replace(/^#\s*/, '');
            if (!text || !packageWarehouseMap[text]) return;
            if (span.parentElement.querySelector('.c_wh_tag')) return;

            const tag = document.createElement('span');
            tag.className = 'c_wh_tag';
            tag.innerText = packageWarehouseMap[text];
            tag.style.cssText = 'margin-left:8px;padding:1px 6px;background:#f0f5ff;border:1px solid #adc6ff;color:#2f54eb;border-radius:3px;font-size:11px;vertical-align:middle;';
            span.parentElement.insertBefore(tag, span.nextSibling);
        });
    }

    // GM_xmlhttpRequest 封装为 Promise
    function gmFetch(url, data) {
        return new Promise((resolve, reject) => {
            GM_xmlhttpRequest({
                method: 'POST',
                url,
                headers: { 'Content-Type': 'application/json' },
                data: JSON.stringify(data),
                onload: (resp) => {
                    try { resolve(JSON.parse(resp.responseText)); }
                    catch (e) { reject(e); }
                },
                onerror: (e) => reject(e)
            });
        });
    }

    // 通过库存接口查询SKU信息（标题+库存）
    async function fetchSkuInfo(sku) {
        const params = new URLSearchParams({
            pageNo: '1',
            pageSize: '50',
            searchType: 'skuName',
            searchContent: sku,
            inquireType: '2',
            stockStatus: '',
            isGroup: '',
            orderBy: '',
            desc: '',
            fullCid: '',
            queryDistribution: '1',
            saleState: '',
            zoneId: '',
            openFlag: 'false',
            hideZeroInventorySku: '0',
            warehouseIds: getWarehouseId()
        });
        const resp = await fetch('/api/v1/inventory/pageList.json', {
            method: 'POST',
            headers: { 'Content-Type': 'application/x-www-form-urlencoded', 'clienttype': '1' },
            body: params.toString()
        });
        const json = await resp.json();
        if (json.code === 0 && json.data && json.data.page && json.data.page.rows.length) {
            const item = json.data.page.rows[0];
            return { title: item.title || '--', available: item.available ?? '--', image: item.image || '', skuId: item.skuId || null };
        }
        return null;
    }

            searchType: 'skuName',
            searchContent: skuPrefix,
            inquireType: '1',

            desc: '',

            queryDistribution: '1',
            saleState: '',
            zoneId: '',
            openFlag: 'false',
            hideZeroInventorySku: '0',
            warehouseIds: getWarehouseId()
        });
        const resp = await fetch('/api/v1/inventory/pageList.json', {
            method: 'POST',
            headers: { 'Content-Type': 'application/x-www-form-urlencoded', 'clienttype': '1' },

        }
        return [];

    // 从标题中提取尺寸关键词（如 "200*250"）
    function extractSizeKeyword(title) {
        const match = title.match(/(\d+)\s*[\*xX×]\s*(\d+)/);
        return match ? match[0] : null;
    }

    // 从标题中解析图案和尺寸（宽*高）
    // 标题格式: "暗无天日李印花魔术贴;气球熊;150*170"（宽*高）
    function parseTitleInfo(title) {
        const parts = title.split(';');
        const pattern = parts.length >= 2 ? parts[parts.length - 2].trim() : '';
        const sizeMatch = title.match(/(\d+)\s*[\*xX×]\s*(\d+)/);
        const width = sizeMatch ? parseInt(sizeMatch[1]) : 0;
        const height = sizeMatch ? parseInt(sizeMatch[2]) : 0;
        const area = width * height;
        return { pattern, width, height, area };
    }

    // 推荐排序：有库存优先，异图同尺 > 同图更大 > 异图更大
    // 同尺寸判断：宽和高都完全匹配（不是面积相等）
    function sortRecommendations(list, currentTitle) {
        const current = parseTitleInfo(currentTitle);

        function getPriority(info) {
            const sameSize = info.width === current.width && info.height === current.height;
            const samePattern = info.pattern === current.pattern;
            const largerSize = info.area > current.area;
            if (sameSize && !samePattern) return { level: 1, label: '异图同尺' };
            if (samePattern && largerSize) return { level: 2, label: '同图更大' };
            if (!samePattern && largerSize) return { level: 3, label: '异图更大' };
            return { level: 4, label: '' };
        }

        return list.map(item => {
            const info = parseTitleInfo(item.title);
            const pri = getPriority(info);
            return { ...item, _priority: pri.level, _label: pri.label, _info: info };
        }).sort((a, b) => {
            const aHasStock = a.available > 0 ? 1 : 0;
            const bHasStock = b.available > 0 ? 1 : 0;
            if (aHasStock !== bHasStock) return bHasStock - aHasStock;
            if (a._priority !== b._priority) return a._priority - b._priority;

            // 同图更大 / 异图更大 内部排序
            if (a._priority === 2 || a._priority === 3) {
                const aSameH = a._info.height === current.height ? 0 : 1;
                const bSameH = b._info.height === current.height ? 0 : 1;
                // 高度相等的优先
                if (aSameH !== bSameH) return aSameH - bSameH;
                // 面积越小越靠前
                if (a._info.area !== b._info.area) return a._info.area - b._info.area;
            }

            return (b.available ?? 0) - (a.available ?? 0);
        });
    }

    // 从SKU中提取款号（如 "JS-017-120" → "JS-017"）
    function extractStylePrefix(sku) {
        const parts = sku.split('-');
        if (parts.length >= 2) return parts.slice(0, 2).join('-');
        return sku;
    }

        const resp = await fetch('/api/v1/order/new/pageList.json', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json', 'clienttype': '1' },
            body: JSON.stringify({
                searchType: 'variationSku',
                searchContent: sku,
                inquireType: 2,
                pageWarehouseIds: [getWarehouseId()],
                statusList: ['new'],
                allOrder: true,
                historyOrder: 0,
                pageNo: 1,
                pageSize: 300,
                desc: 1,
                orderBy: 'printTime'
            })
        });
        const json = await resp.json();
        if (json.code === 0 && json.data?.page?.rows?.length) {
            return json.data.page.rows.filter(r => r.platformOrderId !== excludeOrderNo);
        }
        return [];
    }

        `;
    }

                                    warehouseId: getWarehouseId(),
                                    skuId: String(targetInfo.skuId),
                                    platformSku: targetSku,
                                    changeType: 'only'
                                });
                                const bResp = await fetch('/api/v1/inventory/mappingRelation/changeMappingRelation.json', {
                                    method: 'POST',
                                    headers: { 'Content-Type': 'application/x-www-form-urlencoded', 'clienttype': '1' },
                                    body: bParams.toString()
                                });
                                const bJson = await bResp.json();
                                if (bJson.code === 0) {

    // 取消换货核心逻辑（不带UI交互，返回 {success, msg}）
    async function doCancelSwap(orderNo, originalSku, targetSku) {
        try {
            const originalInfo = await fetchSkuInfo(originalSku);
            if (!originalInfo || !originalInfo.skuId) {
                return { success: false, msg: '未找到原SKU信息' };
            }

            const orderResp = await fetch('/api/v1/order/new/pageList.json', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json', 'clienttype': '1' },
                body: JSON.stringify({
                    searchType: 'orderNo',
                    searchContent: orderNo,
                    inquireType: 2,
                    allOrder: true,
                    historyOrder: 0,
                    pageNo: 1,
                    pageSize: 10,
                    desc: 1,
                    orderBy: 'printTime'
                })
            });
            const orderJson = await orderResp.json();
            if (orderJson.code !== 0 || !orderJson.data?.page?.rows?.length) {
                return { success: false, msg: '未找到订单' };
            }
            const orderId = orderJson.data.page.rows[0].id;

            const detailResp = await fetch(`/api/v1/order/detail.json?id=${orderId}&historyOrder=0&viewBuyerMessage=false`, {
                headers: { 'clienttype': '1' }
            });
            const detailJson = await detailResp.json();
            if (detailJson.code !== 0 || !detailJson.data?.orderDetail?.orderItemVoList?.length) {
                return { success: false, msg: '获取订单详情失败' };
            }
            const orderItem = detailJson.data.orderDetail.orderItemVoList.find(
                item => item.varSku === targetSku || item.inventorySku === targetSku
            );
            if (!orderItem) {
                return { success: false, msg: '未找到目标SKU对应商品项' };
            }

            const swapParams = new URLSearchParams({
                orderId: String(orderId),
                orderItemId: String(orderItem.id),
                orderState: 'new',
                warehouseId: getWarehouseId(),
                skuId: String(originalInfo.skuId),
                platformSku: originalSku,
                changeType: 'only'
            });
            const swapResp = await fetch('/api/v1/inventory/mappingRelation/changeMappingRelation.json', {
                method: 'POST',
                headers: { 'Content-Type': 'application/x-www-form-urlencoded', 'clienttype': '1' },
                body: swapParams.toString()
            });
            const swapJson = await swapResp.json();
            if (swapJson.code === 0) {
                await deleteSwapRecord(orderNo, originalSku);
                return { success: true, msg: '成功' };
            } else {
                return { success: false, msg: swapJson.msg || '换货接口失败' };
            }
        } catch (e) {
            return { success: false, msg: e.message };
        }
    }

    // 取消换货：恢复原SKU映射并删除云端记录
    async function cancelSwap(orderNo, originalSku, targetSku, resultDiv) {
        if (!confirm('确认取消换货？将恢复为原SKU映射')) return;

        const result = await doCancelSwap(orderNo, originalSku, targetSku);
        if (result.success) {
            alert('已取消换货，恢复为原SKU');
            resultDiv.remove();
        } else {
            alert('取消换货失败: ' + result.msg);
        }
    }

    // 批量取消换货按钮注入
    function injectBatchCancelBtn() {
        // 已存在则跳过
        if (document.querySelector('.c_batch_cancel_swap_btn')) return;
        // 找到批量操作栏第3个按钮
        const refBtn = document.querySelector('#CeilingBox > div.rows.flex.star_align > div.lt > div > div > span:nth-child(3) > button');

    // 批量取消换货逻辑
    async function batchCancelSwap() {
        // 获取所有已勾选的订单行checkbox
        const checkboxes = document.querySelectorAll('#\\30  > div.list_items_out > div > div.lt > label > input[type=checkbox]');
        // 实际上需要获取列表中每行的checkbox，上面那个是全选，我们找每行的
        const rowCheckboxes = document.querySelectorAll('tr td input[type=checkbox]:checked');
        // 收集已勾选行的订单号
        const checkedOrderNos = [];
        rowCheckboxes.forEach(cb => {
            const row = cb.closest('tr');
            if (!row) return;
            const oSpan = row.querySelector('td.odn_el div.copy_txt_box > div > span');
            if (oSpan) {
                const no = oSpan.innerText.trim();
                if (no && /^\d+$/.test(no)) checkedOrderNos.push(no);
            }
        });

        if (!checkedOrderNos.length) {
            alert('未勾选任何订单，或未找到订单号');
            return;
        }

        // 查询云端换货记录
        const records = await querySwapRecords(checkedOrderNos);
        if (!records.length) {
            alert('勾选的订单中没有换货记录');
            return;
        }

        // 去重，同orderNo+originalSku只取一条
        const uniqueRecords = [];
        const seen = new Set();
        records.forEach(r => {
            const key = r.orderNo + '_' + r.originalSku;
            if (!seen.has(key)) { seen.add(key); uniqueRecords.push(r); }
        });

        if (!confirm(`确认批量取消换货？将对 ${uniqueRecords.length} 条记录恢复原SKU映射`)) return;

        let success = 0, fail = 0;
        const btn = document.querySelector('.c_batch_cancel_swap_btn');
        const originalText = btn.innerText;

        for (let i = 0; i < uniqueRecords.length; i++) {
            const r = uniqueRecords[i];
            btn.innerText = `批量取消中 ${i + 1}/${uniqueRecords.length}...`;
            btn.disabled = true;
            const result = await doCancelSwap(r.orderNo, r.originalSku, r.targetSku);
            if (result.success) {
                success++;
                // 移除页面上对应的换货结果卡片
                document.querySelectorAll('tr').forEach(row => {
                    const oSpan = row.querySelector('td.odn_el div.copy_txt_box > div > span');
                    if (oSpan && oSpan.innerText.trim() === r.orderNo) {
                        const card = row.querySelector('.c_swap_result');
                        if (card) card.remove();
                    }
                });
            } else {
                fail++;
                console.log('[Bigseller助手] 批量取消失败:', r.orderNo, r.originalSku, result.msg);
            }
        }

        btn.innerText = originalText;
        btn.disabled = false;
        alert(`批量取消完成！成功: ${success}，失败: ${fail}`);
    }

    // 页面加载时从云端查询换货记录并注入显示
    let _swapRetryCount = 0;
    async function loadSwapRecordsForPage() {
        // 收集当前页所有订单号
        const orderSpans = document.querySelectorAll('td.odn_el div.copy_txt_box > div > span');
        const orderNos = [];
        orderSpans.forEach(span => {
            const no = span.innerText.trim();
            if (no && /^\d+$/.test(no)) orderNos.push(no);
        });
        console.log('[Bigseller助手] 收集到订单号:', orderNos.length, JSON.stringify(orderNos));
        if (!orderNos.length) return;

        const records = await querySwapRecords(orderNos);
        console.log('[Bigseller助手] 云端返回换货记录:', records.length, JSON.stringify(records.map(r => ({ orderNo: r.orderNo, originalSku: r.originalSku }))));
        if (!records.length) return;

        // 按订单号分组（去重，同orderNo+originalSku只取最新一条）
        const recordMap = {};
        records.forEach(r => {
            if (!recordMap[r.orderNo]) recordMap[r.orderNo] = [];
            const exists = recordMap[r.orderNo].find(x => x.originalSku === r.originalSku);
            if (!exists) recordMap[r.orderNo].push(r);
        });
        console.log('[Bigseller助手] 记录分组keys:', JSON.stringify(Object.keys(recordMap)));
        // 对比：哪些订单号在记录里有但页面收集时没有
        const missingOnPage = Object.keys(recordMap).filter(k => !orderNos.includes(k));
        if (missingOnPage.length) console.log('[Bigseller助手] 云端有记录但页面未收集到的订单号:', JSON.stringify(missingOnPage));

        // 遍历页面，为有换货记录的SKU注入结果
        const allRows = document.querySelectorAll('tr');
        console.log('[Bigseller助手] 页面tr数量:', allRows.length);
        let matched = 0;
        const foundOrderNos = new Set();
        allRows.forEach(row => {
            const oSpan = row.querySelector('td.odn_el div.copy_txt_box > div > span');
            if (!oSpan) return;
            const orderNo = oSpan.innerText.trim();
            if (!recordMap[orderNo]) return;
            foundOrderNos.add(orderNo);

            const skuSpan = row.querySelector('td.merge_el.dt_el div.copy_txt_box > div > span');
            if (!skuSpan) {
                console.log('[Bigseller助手] 订单', orderNo, '找到了odn_el但没找到merge_el.dt_el的SKU span');
                return;
            }
            const sku = skuSpan.innerText.trim();

            const match = recordMap[orderNo].find(r => r.originalSku === sku);
            if (!match) {
                console.log('[Bigseller助手] 订单', orderNo, '有记录但SKU不匹配, 页面SKU:', sku, '记录SKU:', recordMap[orderNo].map(r => r.originalSku));
                return;
            }

            const skuCell = skuSpan.closest('td.merge_el.dt_el');
            if (!skuCell || skuCell.querySelector('.c_swap_result')) return;

            matched++;
            const resultDiv = document.createElement('div');
            resultDiv.className = 'c_swap_result';
            resultDiv.style.cssText = 'margin-top:8px;padding:8px;border:1px solid #52c41a;border-radius:6px;background:#f6ffed;position:relative;';
            resultDiv.innerHTML = `
                <div style="text-align:center;color:#52c41a;font-size:14px;">↓</div>
                <div style="display:flex;align-items:center;gap:8px;">
                    <img src="${match.targetImage}" style="width:40px;height:40px;object-fit:cover;border-radius:4px;${match.targetImage ? '' : 'display:none;'}" />
                    <div style="flex:1;">
                        <div style="font-size:13px;font-weight:bold;word-break:break-all;">${match.targetSku}</div>
                        <div style="font-size:11px;color:#666;word-break:break-all;">${match.targetTitle}</div>
                        <div style="font-size:11px;color:#52c41a;">库存: ${match.targetAvailable}</div>
                    </div>
                    <button class="c_cancel_swap_btn" style="padding:2px 8px;border:1px solid #ff4d4f;background:#fff;color:#ff4d4f;border-radius:3px;cursor:pointer;font-size:11px;white-space:nowrap;">取消换货</button>
                </div>
            `;
            skuCell.appendChild(resultDiv);

            resultDiv.querySelector('.c_cancel_swap_btn').onclick = () => {
                cancelSwap(orderNo, match.originalSku, match.targetSku, resultDiv);
            };
        });
        // 汇总：哪些有记录的订单号在tr遍历中没被匹配到
        const notFoundInRows = Object.keys(recordMap).filter(k => !foundOrderNos.has(k));
        if (notFoundInRows.length) console.log('[Bigseller助手] 有记录但遍历tr时未找到对应行的订单号:', JSON.stringify(notFoundInRows));
        console.log('[Bigseller助手] 成功注入换货结果:', matched, '/', Object.keys(recordMap).length);

        // 如果有记录但匹配为0，说明DOM还没渲染完，延迟重试（最多重试5次）
        if (matched === 0 && Object.keys(recordMap).length > 0 && _swapRetryCount < 5) {
            _swapRetryCount++;
            console.log('[Bigseller助手] DOM未就绪，' + _swapRetryCount + '秒后重试 (' + _swapRetryCount + '/5)');
            setTimeout(loadSwapRecordsForPage, _swapRetryCount * 1000);
        } else if (matched > 0) {
            _swapRetryCount = 0;
        }
    }

    // 初始化：获取仓库列表
    async function init() {
        warehouseList = await fetchWarehouseList();
        if (warehouseList.length) {
            // 如果没有记忆或记忆的仓库不在列表中，使用默认仓库
            const savedId = localStorage.getItem(WH_STORAGE_KEY);
            const validSaved = warehouseList.find(w => w.id === savedId);
            if (validSaved) {
                currentWarehouseId = savedId;
            } else {
                const defaultWh = warehouseList.find(w => w.isDefault) || warehouseList[0];
                currentWarehouseId = defaultWh.id;
                localStorage.setItem(WH_STORAGE_KEY, currentWarehouseId);
            }
        }
    }
    init();

    // 初始注入 + 监听DOM变化（应对SPA翻页/筛选）
    const observer = new MutationObserver(() => injectButtons());
    observer.observe(document.body, { childList: true, subtree: true });
    setTimeout(injectButtons, 1500);

    // 等待订单列表渲染后查询云端换货记录（防抖机制，DOM变化时重新检测）
    let swapDebounceTimer = null;
    let lastOrderCount = 0;
    const swapObserver = new MutationObserver(() => {
        const orderSpans = document.querySelectorAll('td.odn_el div.copy_txt_box > div > span');
        if (orderSpans.length > 0 && orderSpans.length !== lastOrderCount) {
            lastOrderCount = orderSpans.length;
            clearTimeout(swapDebounceTimer);
            swapDebounceTimer = setTimeout(() => {
                loadSwapRecordsForPage();
                injectWarehouseTags();
            }, 800);
        }
    });
    swapObserver.observe(document.body, { childList: true, subtree: true });
})();
